-- Table: states
CREATE TABLE states (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL
);

-- Table: cities
CREATE TABLE cities (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    pincode VARCHAR(10),
    state_id INT,
    FOREIGN KEY (state_id) REFERENCES states(id)
);

-- Table: customers
CREATE TABLE customers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    fullname VARCHAR(100),
    dob DATE,
    age INT,
    state VARCHAR(100),
    city VARCHAR(100),
    pincode VARCHAR(10)
);
